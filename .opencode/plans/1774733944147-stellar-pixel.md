## 4. Replace `_install_global_popup()` (lines 353–395)

**Replace entire method body with:**

```python
def _install_global_popup(self) -> None:
    if ContextMenu is None:
        self.set_status("ContextMenu unavailable")
        return
    try:
        self.context_menu = ContextMenu(self, scale=0.9, width=220)
        self.context_menu.add_option(option="Open Input Dialog", command=self.show_input_dialog, accelerator="CmdOrCtrl+O")
        self.context_menu.add_option(option="Start Progress Popup", command=self.start_progress_popup, accelerator="CmdOrCtrl+P")
        self.context_menu.add_option(option="Toggle Loader", command=self.toggle_loader, accelerator="CmdOrCtrl+L")
        self.context_menu.add_separator()
        self.context_menu.add_option(option="Theme Info", command=self.show_theme_info)
        self.set_status("Context menu enabled (right-click)")
    except Exception as exc:
        warn_import("ContextMenu runtime", exc)
```

**Important:** Remove the `bind_all("<Button-3>")` and `bind_all("<Button-2>")` calls — `ContextMenu` handles binding automatically.

**Note on scope:** `ContextMenu(self, ...)` binds to `self` (the CTk root window) and its children, which gives the same "right-click anywhere" behavior.

## 5. New Helper Methods

Add these after `_install_global_popup()`:

### `_on_appearance(mode: str)` — appearance toggle with mutual exclusion
```python
def _on_appearance(self, mode: str) -> None:
    self.set_appearance(mode)
    self._sync_appearance_checks(mode)

def _sync_appearance_checks(self, mode: str) -> None:
    for item_mode, item_ref in [
        ("Dark", self.appearance_dark_item),
        ("Light", self.appearance_light_item),
        ("System", self.appearance_system_item),
    ]:
        if item_ref is not None:
            item_ref.set_checked(item_mode == mode)
```

### Recent files helpers
```python
def _open_recent(self, filename: str) -> None:
    self.set_status(f"Open recent: {filename}")

def _open_most_recent(self) -> None:
    if self.recent_option_items:
        self.set_status("Opening most recent file (demo)")
    else:
        self.set_status("No recent files")

def _clear_recent_items(self) -> None:
    if self.recent_menu is None:
        return
    for item in self.recent_option_items:
        try:
            self.recent_menu.remove_option(item)
        except Exception:
            pass
    self.recent_option_items.clear()
    if self.open_most_recent_item is not None:
        self.open_most_recent_item.enable(False)
    self.set_status("Recent files cleared (remove_option)")

def _reset_recent_menu(self) -> None:
    if self.recent_menu is None:
        return
    self.recent_menu.clean()
    self.recent_option_items.clear()
    self._seed_recent_items()
    self.set_status("Recent files reset (clean + re-seed)")

def _seed_recent_items(self) -> None:
    if self.recent_menu is None:
        return
    demo_files = ["artwork_01.lines", "logo_v2.lines", "icon_set.lines"]
    for f in demo_files:
        item = self.recent_menu.add_option(
            option=f, command=lambda fn=f: self._open_recent(fn)
        )
        self.recent_option_items.append(item)
    self.recent_menu.add_separator()
    self.recent_menu.add_option(option="Clear Recent", command=self._clear_recent_items)
    self.recent_menu.add_option(option="Reset Recent", command=self._reset_recent_menu)
    if self.open_most_recent_item is not None:
        self.open_most_recent_item.enable(True)

def _toggle_open_recent_enabled(self) -> None:
    if self.open_most_recent_item is None:
        return
    try:
        # Toggle: if currently disabled, enable; if enabled, disable
        self.open_most_recent_item.enable(not self.open_most_recent_item.enabled)
        state = "enabled" if self.open_most_recent_item.enabled else "disabled"
        self.set_status(f"'Open Most Recent' is now {state}")
    except Exception:
        self.set_status("Could not toggle 'Open Most Recent'")
```

## 7. _build_menu_bar() Detailed Build Sequence

```python
def _build_menu_bar(self) -> None:
    if CTkMenuBar is None or CustomDropdownMenu is None:
        fallback = ctk.CTkFrame(self, corner_radius=8)
        fallback.pack(fill=tk.X, padx=10, pady=(10, 6))
        ctk.CTkLabel(fallback, text="CTkMenuBar not available. Menus are disabled.", anchor="w").pack(fill=tk.X, padx=10, pady=8)
        return

    self.menu_bar = CTkMenuBar(self, bg_color=["#f8f8fa", "#17171a"])
    self.menu_bar.pack(fill=tk.X, padx=10, pady=(10, 4))

    file_btn = self.menu_bar.add_cascade(text="File")
    ui_font = file_btn.cget("font")

    # --- File Menu ---
    file_menu = CustomDropdownMenu(widget=file_btn, width=220, font=ui_font)
    file_menu.add_option(option="Open Input Dialog", command=self.show_input_dialog, accelerator="CmdOrCtrl+O")
    file_menu.add_option(option="Open Toplevel", command=self.open_toplevel_demo, accelerator="CmdOrCtrl+Shift+N")
    file_menu.add_separator()
    export_sub = file_menu.add_submenu("Export As", accelerator="CmdOrCtrl+Shift+E")
    export_sub.add_option(option="PDF", command=lambda: self.set_status("Export: PDF"), accelerator="CmdOrCtrl+Alt+P")
    export_sub.add_option(option="SVG", command=lambda: self.set_status("Export: SVG"), accelerator="CmdOrCtrl+Alt+S")
    self.recent_menu = file_menu.add_submenu("Recent Files")
    self._seed_recent_items()
    self.open_most_recent_item = file_menu.add_option(
        option="Open Most Recent", command=self._open_most_recent,
        accelerator="CmdOrCtrl+Shift+R", enabled=False,
    )
    file_menu.add_separator()
    file_menu.add_option(option="Quit", command=self.on_close, accelerator="CmdOrCtrl+Q")

    # --- Edit Menu ---
    edit_btn = self.menu_bar.add_cascade(text="Edit")
    edit_menu = CustomDropdownMenu(widget=edit_btn, width=220, font=ui_font)
    edit_menu.add_option(option="Clear Status", command=lambda: self.set_status("Ready"), accelerator="CmdOrCtrl+K")
    edit_menu.add_option(option="Progress Popup", command=self.start_progress_popup, accelerator="CmdOrCtrl+P")
    edit_menu.add_option(option="Toggle Loader", command=self.toggle_loader, accelerator="CmdOrCtrl+L")
    edit_menu.add_separator()
    edit_menu.add_option(option='Toggle "Open Recent"', command=self._toggle_open_recent_enabled, accelerator="CmdOrCtrl+Alt+R")

    # --- View Menu ---
    view_btn = self.menu_bar.add_cascade(text="View")
    view_menu = CustomDropdownMenu(widget=view_btn, width=220, font=ui_font)
    appearance_sub = view_menu.add_submenu("Appearance")
    self.appearance_dark_item = appearance_sub.add_option(
        option="Dark", command=lambda: self._on_appearance("Dark"),
        accelerator="CmdOrCtrl+Shift+D", checkable=True, checked=False,
    )
    self.appearance_light_item = appearance_sub.add_option(
        option="Light", command=lambda: self._on_appearance("Light"),
        accelerator="CmdOrCtrl+Shift+L", checkable=True, checked=False,
    )
    self.appearance_system_item = appearance_sub.add_option(
        option="System", command=lambda: self._on_appearance("System"),
        accelerator="CmdOrCtrl+Shift+S", checkable=True, checked=True,
    )
    view_menu.add_separator()
    view_menu.add_option(option="Theme Info", command=self.show_theme_info, accelerator="CmdOrCtrl+I")

    # --- Help Menu (scrollable) ---
    help_btn = self.menu_bar.add_cascade(text="Help")
    help_menu = CustomDropdownMenu(
        widget=help_btn, width=280, font=ui_font,
        max_visible_options=6, enable_scrollbar=True,
    )
    help_menu.add_option(option="About ThemeManager", command=self.show_theme_info, accelerator="F1")
    help_menu.add_option(option="About Async", command=lambda: self.set_status("See Async Demo tab"))
    help_menu.add_separator()
    help_menu.add_option(option="Accelerators Demo", command=lambda: self.set_status("Accelerators bind real shortcuts"))
    help_menu.add_option(option="Checkable Items Demo", command=lambda: self.set_status("View → Appearance uses checkable items"))
    help_menu.add_option(option="Scrollable Menu Demo", command=lambda: self.set_status("This Help menu is scrollable (max_visible_options=6)"))
    help_menu.add_option(option="Context Menu Demo", command=lambda: self.set_status("Right-click anywhere for ContextMenu"))
    help_menu.add_option(option="Dynamic Enable Demo", command=lambda: self.set_status("File → Open Most Recent starts disabled"))
    help_menu.add_option(option="Remove/Clean Demo", command=lambda: self.set_status("File → Recent Files → Clear/Reset"))
```
